select 
	a.ID as ID,
	a.COMANDO as COMANDO,
	a.NOME as NOME_PAINEL,
	a.LINK as LINK,
	b.CHAT_ID as CHAT_ID_GRUPO,
	b.NOME as NOME_GRUPO,
	b.OPERACAO as VERTICAL_NEGOCIO,
	d.HORA as HORA_REPORT,
	e.NOME as NOME_ROTINA,
	e.EXPRESSAO_CRON as EXPRESSAO_CRON,
	a.TIPO_ENVIO as TIPO_ENVIO,
	a.ATIVO as ATIVO,
	a.GERANDO as GERANDO
from argos.t_painel_automation a
	join argos.t_grupo_telegram b on b.CHAT_ID = a.CHAT_ID 
	join argos.t_divulgacao_automation c on c.PAINEL_ID = a.ID  
	join argos.t_horario_divulgacao_automation d on d.ID = c.HORA_DIVULGACAO_ID 
	join argos.t_trabalho_divulgacao_automation e on e.ID = d.ID_TRABALHO  
where a.ID = ':ID' 
and d.HORA = ':HORA_ENVIO';
