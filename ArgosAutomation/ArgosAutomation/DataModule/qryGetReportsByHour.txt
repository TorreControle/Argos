select 
	a.ID as ID,
	a.NOME as NOME,
	b.NOME as NOME_GRUPO_TELEGRAM,
	b.CHAT_ID as CHAT_ID_GRUPO,
	b.OPERACAO OPERACAO,
	a.GERANDO as GERANDO
from argos.t_painel_automation a
	join argos.t_grupo_telegram b on a.CHAT_ID = b.CHAT_ID
	join argos.t_divulgacao_automation c on c.PAINEL_ID = a.ID 
	join argos.t_horario_divulgacao_automation d on d.ID = c.HORA_DIVULGACAO_ID
	join argos.t_trabalho_divulgacao_automation e on e.ID = d.ID
where d.HORA = ':HORA_DIVULGACAO';
