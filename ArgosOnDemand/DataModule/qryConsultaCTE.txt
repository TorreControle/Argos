SELECT
  DISTINCT(NUM_CONHECIMENTO) AS NUM_CONHECIMENTO,
  SERIE,
  NOTAS_FISCAIS,
  CANCELADA,
  UNIDADE_NEGOCIO,
  PRESTADOR_SERVICO_CNPJ,
  PRESTADOR_SERVICO_RAZAO_SOCIAL AS PRESTADOR_SERVICO,
  COB_CNPJ_CPF AS DESTINATARIO_CNPJ_CPF,
  COB_RAZAO_SOCIAL AS DESTINATARIO,
  concat(SIGLA_UF_COLETA, ' - ', NOME_LOCAL_COLETA) AS LOCAL_COLETA,
  concat(SIGLA_UF_ENTREGA, ' - ', NOME_LOCAL_ENTREGA) AS LOCAL_ENTREGA,
  DATA_EMISSAO AS DATA_EMISSAO_ROMANEIO,
  NUM_ROMANEIO AS NUM_ROMANEIO,
  PLACA_TRACAO
FROM
  hive_metastore.ouro_transportes.fato_cte
WHERE
  NUM_CONHECIMENTO  = ':CTE'
  AND NUM_ROMANEIO = ':ROMANEIO'
